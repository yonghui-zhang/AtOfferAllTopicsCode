#include<stdio.h>
#include<string.h>
/****删除字符串中的重复字符***/
void deleteStringRepeatChar(char str[])
{
	if(str == NULL)
	{
		return ;
	}
	/****初始化256个桶****/
	int ascii[256];
	for(int i=0;i<256;i++)
	{
        ascii[i] = 0;
	}
	/****遍历字符串进桶和判断******/
    for(i=0;i<strlen(str);i++)
	{
		if(ascii[str[i]] == 0)
		{
           ascii[str[i]]++;
		   printf("%c", str[i]);
		}
	}
	printf("\n");
}
int main()
{
	char str[] = {"google"};
	deleteStringRepeatChar(str);
}
